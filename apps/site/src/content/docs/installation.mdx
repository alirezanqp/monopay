import { Tabs, TabItem } from '@astrojs/starlight/components';

To install monopay, simply use your package manager of choice and install it

<Tabs>
    <TabItem label="npm">
    ```shell
    npm install monopay
    ```
    </TabItem>
    <TabItem label="yarn">
    ```shell
    yarn add monopay
    ```
    </TabItem>
    <TabItem label="pnpm">
    ```shell
    pnpm add monopay
    ```
    </TabItem>
    <TabItem label="bun">
    ```shell
    bun add monopay
    ```
    </TabItem>
</Tabs>
